P.S: Распознаваемые команды:

  "turn_on": ["object"],		(включить синий/красный/зелёный свет, вентилятор, музыку)
  "turn_off": ["object"],		(выключить свет, вентилятор, музыку)
  "timer": ["duration_num", "duration_unit"],	(таймер на какое-то время)
  "alarm": ["time"],    (будильник)
  "volume":	(увеличить, уменьшить громкость)
  "search":	(поиск в firefox, можно заменить на гугл например)
  "temperature"	(температура)
  "time":		(время сейчас)
  "humidity":	(влажность)
  "sound": 	(заглушить звук)
  "thanks":   	(благодарность)
  "news": 	(читать новости)
  "token": 	(очистка сессии если gptAPI)
  "work": 	(рабочий режим)
  "block": 	(блокировать систему)
  "telesend": 	(отправить что либо)
  "raspconnect": 	(подключение к серверу for Ex.)
